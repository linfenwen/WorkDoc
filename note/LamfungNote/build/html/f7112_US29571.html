HWND CSmRecordSessionMgr::FindRecordWindow()
{
    HWND hWnd = ::FindWindowEx(NULL, NULL, _T("#32770"), NULL);
    do
    {
        HWND hChildWnd = ::FindWindowEx(hWnd, NULL, NULL, _T("0xRecordPanleID853"));
        if(hChildWnd != NULL)
        {
            return hWnd;
        }
        hWnd = ::FindWindowEx(NULL, hWnd, _T("#32770"), NULL);
    }
    while(hWnd != NULL);

    return NULL;
}
